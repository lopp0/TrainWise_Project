שורות 37–60 — טעינת session שמור 
useEffect(() => {
  (async () => {
    try {
      const [userJson, storedDeviceId] = await Promise.all([
        AsyncStorage.getItem(KEYS.USER),
        AsyncStorage.getItem(KEYS.DEVICE_ID),
      ]);
      if (userJson) {
        const parsed = JSON.parse(userJson);
        setUser(parsed);
        setUserId(parsed.userId ?? parsed.id ?? null);
      }
      if (storedDeviceId) setDeviceId(storedDeviceId);
    } catch {}
    finally { setReady(true); }
  })();
}, []);
ה-useEffect רץ פעם אחת בלבד (מערך תלויות ריק []) מיד אחרי ה-render הראשון. Promise.all קורא את שני המפתחות במקביל — יעיל יותר מ-await אחד אחרי השני. אם יש JSON שמור — מפענחים ומעדכנים state. finally מבטיח ש-ready=true תמיד, גם אם הייתה שגיאה (try/catch ריק בולע אותה).
שורות 62–90 — פונקציית login 
const login = async (email, password) => {
  const res = await api.login(email, password);
  const userData = res.data;
  let dId = deviceId;
  if (!dId) {
    dId = 'device-' + Date.now();
    await AsyncStorage.setItem(KEYS.DEVICE_ID, dId);
    setDeviceId(dId);

שורות 92–102 — פונקציית logout 
const logout = async () => {
  await AsyncStorage.removeItem(KEYS.USER);
  setUser(null);
  setUserId(null);
};
logout מוחקת רק את פרטי המשתמש — לא את ה-deviceId. כך בהתחברות הבאה המכשיר עדיין מזוהה. ה-state מתאפס ל-null, מה שיגרום לניווט חזרה למסך ההתחברות.
שורות 104–110 — updateUser 
const updateUser = (patch) => {
  const merged = { ...user, ...patch };
  setUser(merged);
  AsyncStorage.setItem(KEYS.USER, JSON.stringify(merged));
};
updateUser מאפשרת לעדכן שדות ספציפיים במשתמש מבלי לרענן מהשרת. Spread operator { ...user, ...patch } ממזג — ערכי patch מדרסים ערכים קיימים. השמירה ב-AsyncStorage מבטיחה שהעדכון יישמר גם אחרי סגירת האפליקציה.

CurvedText — טקסט על מעגל 
const CurvedText = ({ text, radius }) => {
  const chars = text.split('');
  const angleStep = 360 / chars.length;
  return chars.map((ch, i) => {
    const angle = (i * angleStep - 90) * (Math.PI / 180);
    const x = radius * Math.cos(angle);
    const y = radius * Math.sin(angle);
    return (
      <Text key={i} style={{
        position:'absolute',
        left: x + radius,
        top:  y + radius,
        transform:[{ rotate: `${i * angleStep}deg` }],
      }}>{ch}</Text>
    );
  });
};
CurvedText — טכניקה גאומטרית לפיזור תווים על מעגל. angleStep = 360 חלקי מספר תווים — זווית בין כל תו. -90 — מתחיל מלמעלה (12 בשעון) ולא מימין (3 בשעון). Math.PI/180 — המרת דרגות לרדיאנים (JavaScript עובד ברדיאנים). Math.cos / Math.sin — חישוב x,y על המעגל. כל תו ממוקם בposition absolute ומסובב כדי "לפנות" החוצה מהמעגל.

חישוב ACWR 
const acute   = last7DaysLoad;           // סכום 7 ימים
const chronic = last28DaysLoad / 4;      // ממוצע שבועי על 28 יום
const acRatio = chronic > 0
  ? acute / chronic : 0;
ACWR — Acute-to-Chronic Workload Ratio. מדד מחקרי (Gabbett, 2016) לסיכון פציעה. Acute = עומס שבוע אחרון (7 ימים). Chronic = ממוצע שבועי של 28 ימים אחרונים. AC Ratio = Acute / Chronic. Zone בטוחה: 0.8–1.3. מתחת ל-0.8: undertraining. מעל 1.5: סיכון גבוה לפציעה.

src/theme/colors.js

Singleton ניתן-לשינוי + Fonts + Spacing
Colors Singleton 
export const Colors = { ...darkPalette };
let _activeTheme = 'dark';
const _listeners = new Set();
Singleton — אובייקט יחיד בכל הזיכרון. כל קובץ שמייבא Colors מקבל הפניה לאותו אובייקט. Spread ({ ...darkPalette }) — יוצר עותק של ה-darkPalette ב-Colors. _listeners — Set של callback functions שיקבלו הודעה בשינוי תמה.
applyTheme 
export const applyTheme = (themeName) => {
  const palette = PALETTES[themeName] || darkPalette;
  Object.keys(Colors).forEach(k => delete Colors[k]);
  Object.assign(Colors, palette);
  _activeTheme = palette === lightPalette ? 'light' : 'dark';
  _listeners.forEach(fn => fn(_activeTheme));
};
מנגנון חכם: במקום ליצור אובייקט חדש (שישבור import references), אנחנו: 1. מוחקים את כל המפתחות הישנים. 2. מכניסים את המפתחות החדשים. כך כל קובץ שייבא Colors בעבר עדיין מצביע לאותו אובייקט — אבל עם ערכים חדשים! לאחר העדכון — מודיע לכל המאזינים.

src/theme/ThemeContext.js

Context לניהול תמה עם שמירה קבועה Context
ThemeProvider 
export const ThemeProvider = ({ children }) => {
  const [theme, setThemeState] = useState(getActiveTheme());

  useEffect(() => {
    (async () => {
      const saved = await AsyncStorage.getItem(STORAGE_KEY);
      if (saved === 'light' || saved === 'dark') {
        applyTheme(saved);
        setThemeState(saved);
      }
    })();
  }, []);
בעת עלייה — קורא את התמה השמורה מ-AsyncStorage. getActiveTheme() נותן את הערך ה-in-memory הנוכחי (dark כברירת מחדל). async IIFE (Immediately Invoked Function Expression) — פונקציה שקוראת לעצמה מיד, מאפשרת async בתוך useEffect.
setTheme + שמירה 
const setTheme = useCallback(async (next) => {
  const safe = next === 'light' ? 'light' : 'dark';
  applyTheme(safe);
  setThemeState(safe);
  await AsyncStorage.setItem(STORAGE_KEY, safe);
}, []);
useCallback עם מערך תלויות ריק — פונקציה נוצרת פעם אחת בלבד. safe — ולידציה: רק 'light' מקובל; כל שאר הערכים הופכים ל-'dark'. 3 פעולות: עדכון Colors singleton + עדכון state + שמירה קבועה.
React.Fragment key trick 
<React.Fragment key={theme}>{children}</React.Fragment>
טריק מתוחכם: כש-key משתנה, React מפרק ומרכיב מחדש את כל הצאצאים. זה מבטיח שמסכים שקראו Colors.background בזמן render (ולא דרך Context) יקבלו את הצבעים החדשים. עלות: re-mount מלא — מקובל כי קורה רק בלחיצת המשתמש, לא אוטומטית.
useTheme Hook 
export const useTheme = () => useContext(ThemeContext);
שורה אחת שמפשטת גישה: const {'{'}theme, setTheme{'}'} = useTheme() בכל קומפוננט. ללא ה-Hook, כל קומפוננט היה צריך: import ThemeContext; useContext(ThemeContext).
